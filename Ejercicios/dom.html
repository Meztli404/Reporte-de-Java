const myDiv3 = document.getElementById('myDiv3');
// Manejar el evento click

myDiv3.addEventListener('click', () => {
alert('Div clicked!');
myDiv3.style.backgroundColor = 'lightblue';
});
//Manejar el evento mouseover
myDiv3.addEventListener('mouseover', () => {
myDiv3.style.backgroundColor = 'yellow';
});

//Manejar el evento mouseout
myDiv3.addEventListener('mouseout', () => {
myDiv3.style.backgroundColor = '';
});

// Cambiar el contenido del título y los párrafos
document.getElementById('changeContent').addEventListener('click', () => {
document.getElementById('title').textContent = 'Updated Web Page';
const paragraphs = document.getElementsByClassName('text');
for (let p of paragraphs) {
p.textContent = 'This paragraph has been updated.';
}
});

// Resaltar el texto de los párrafos
document.getElementById('highlightText').addEventListener('click', () => {
const paragraphs = document.getElementsByClassName('text');
for (let p of paragraphs) {
p.classList.add('highlight');
}
});

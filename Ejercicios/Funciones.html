Division de dos numeros

Funcion Tradicional

                        function dividir(a, b) {
                            if (b === 0) {
                                return "Error: Division por cero";
                            }
                            return a / b;
                        }

Funcion Flecha

                        const dividirFlecha = (a, b) => {
                            if (b === 0) {
                                return "Error: Division por cero";
                            }
                            return a / b;
                        };
